SELECT DISTINCT T1.FACT_ODR_NO,T1.ODR_QTY,T2.ODR_QTY OLD_QTY,T1.STYLE_NO,T1.ARTIC_NO,t1.rev_odr_date,T1.ODR_QTY - T2.ODR_QTY AS UPDATE_QTY 
FROM MDMIS.ODRM T1, MDMIS.TXODRM T2 
WHERE SUBSTRB( t1.modify_date,1,8) between 'frDate' and 'toDate'

AND T2.FACT_NO = T1.FACT_NO AND T2.FACT_ODR_NO= T1.FACT_ODR_NO 
AND T1.ODR_QTY <>T2.ODR_QTY 
AND t1.cancel_date IS NULL
